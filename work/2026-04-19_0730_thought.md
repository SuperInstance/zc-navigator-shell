**Action:** Let me get more detailed analysis with a proper script.

```bash
cd /home/runner/navigator-shell
cat > analyze_readmes.sh << 'EOF'
#!/bin/bash
BASE_DIR="/home/runner/workspace/superinstance"
OUTPUT_FILE="work/2024-03-20_0001_readme_detailed.md"

echo "# SuperInstance README Quality Report" > $OUTPUT_FILE
echo "Generated: $(date)" >> $OUTPUT_FILE
echo "" >> $OUTPUT_FILE
echo "## Scoring Criteria" >> $OUTPUT_FILE
echo "- **Size Score**: 0-3 points (0: <100B, 1: 100-500B, 2: 500B-2KB, 3: >2KB)" >> $OUTPUT_FILE
echo "- **Section Score**: 0-4 points (Installation, Usage, API/Reference, Examples)" >> $OUTPUT_FILE
echo "- **Recency Score**: 0-2 points (2: updated <90 days, 1: 90-365 days, 0: >365 days)" >> $OUTPUT_FILE
echo "- **Total**: 0-9 points" >> $OUTPUT_FILE
echo "" >> $OUTPUT_FILE
echo "## Worst READMEs (by total score)" >> $OUTPUT_FILE
echo "" >> $OUTPUT_FILE
echo "| Repo | Size | Sections | Recency | Total | Path |" >> $OUTPUT_FILE
echo "|------|------|----------|---------|-------|------|" >> $OUTPUT_FILE

find "$BASE_DIR" -name "README.md" -type f | while read readme; do
    # Get size in bytes
    size=$(wc -c < "$readme")
    
    # Size score
    if [ $size -lt 100 ]; then
        size_score=0
    elif [ $size -lt 500 ]; then
        size_score=1
    elif [ $size -lt 2000 ]; then
        size_score=2
    else
        size_score=3
    fi
    
    # Section score
    section_score=0
    content=$(cat "$readme")
    if [[ "$content" =~ [Ii]nstall ]]; then
        ((section_score++))
    fi
    if [[ "$content" =~ [Uu]sage ]]; then
        ((section_score++))
    fi
    if [[ "$content" =~ [Aa][Pp][Ii] ]] || [[ "$content" =~ [Rr]eference ]]; then
        ((section_score++))
    fi
    if [[ "$content" =~ [Ee]xample ]]; then
        ((section_score++))
    fi
    
    # Recency score
    mod_date=$(stat -c %Y "$readme")
    now=$(date +%s)
    days_old=$(( (now - mod_date) / 86400 ))
    
    if [ $days_old -lt 90 ]; then
        recency_score=2
    elif [ $days_old -lt 365 ]; then
        recency_score=1
    else
        recency_score=0
    fi
    
    total=$((size_score + section_score + recency_score))
    
    # Only show worst ones (total <= 3)
    if [ $total -le 3 ]; then
        rel_path=${readme#$BASE_DIR/}
        dir_path=$(dirname "$rel_path")
        echo "| $dir_path | $size_score | $section_score | $recency_score | $total | $rel_path |" >> $OUTPUT_FILE
    fi
done

echo "" >> $OUTPUT_FILE
echo "## Summary" >> $OUTPUT_FILE
echo "Found $(find "$BASE_DIR" -name "README.md" -type f | wc -l) README.md files total." >> $OUTPUT_FILE
echo "Showing $(grep -c "^\|" $OUTPUT_FILE) READMEs with score ≤ 3." >> $OUTPUT_FILE
EOF

chmod +x analyze_readmes.sh
./analyze_readmes.sh
```
