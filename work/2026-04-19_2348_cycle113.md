```bash
#!/bin/bash
score_readme() {
    local readme_path="$1"
    local score=0
    
    # Existence
    if [[ -f "$readme_path" ]]; then
        score=$((score + 1))
        content=$(cat "$readme_path" 2>/dev/null)
        # Length
        len=${#content}
        if (( len > 500 )); then
            score=$((score + 3))
        elif (( len >= 100 )); then
            score=$((score + 2))
        elif (( len > 0 )); then
            score=$((score + 1))
        fi
        # Structure (simple heuristic: count markdown headers)
        headers=$(grep -c "^#" "$readme_path" 2>/dev/null)
        if (( headers >= 3 )); then
            score=$((score + 3))
        elif (( headers >= 1 )); then
            score=$((score + 2))
        else
            score=$((score + 1))
        fi
        # Freshness (check last git commit for README)
        last_commit=$(git -C "$(dirname "$readme_path")" log --oneline -1 -- "$(basename "$readme_path")" 2>/dev/null | head -1)
        if [[ $last_commit =~ 202[5-9] ]]; then
            score=$((score + 2))
        elif [[ $last_commit =~ 202[0-4] ]]; then
            score=$((score + 1))
        fi
        # Clarity (heuristic: check for common technical terms)
        if grep -q -i "install\|usage\|example\|api\|config" "$readme_path" 2>/dev/null; then
            score=$((score + 2))
        else
            score=$((score + 1))
        fi
    fi
    echo $score
}

# Main discovery and scoring
declare -a repo_scores
while IFS= read -r git_dir; do
    repo_dir=$(dirname "$git_dir")
    # Quick SuperInstance filter: check directory name
    if [[ "$repo_dir" =~ (cocapn|superinstance|fleet|plato|holodeck) ]]; then
        for readme in "$repo_dir"/README*; do
            if [[ -f "$readme" ]]; then
                score=$(score_readme "$readme")
                repo_scores+=("$score:$repo_dir")
                break
            fi
        done
    fi
done < <(find /home -type d -name ".git" 2>/dev/null | head -30)

# Sort and output top 10 worst
printf "Score\tRepo Path\n"
printf "=====\t=========\n"
for entry in "${repo_scores[@]}"; do
    echo "$entry"
done | sort -t: -k1n | head -10 | sed 's/:/\\t/'
```
