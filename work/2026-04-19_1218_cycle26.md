**Command 2: Score READMEs (script)**
```bash
#!/bin/bash
score_repo() {
    repo_path="$1"
    readme_file=""
    # Look for README files
    for f in "$repo_path"/README* "$repo_path"/readme*; do
        if [[ -f "$f" ]]; then
            readme_file="$f"
            break
        fi
    done
    
    score=0
    issues=""
    
    if [[ -z "$readme_file" ]]; then
        score=$((score - 2))
        issues="Missing README"
    else
        content=$(cat "$readme_file" 2>/dev/null | tr -d '\0')
        length=${#content}
        
        if [[ $length -lt 50 ]]; then
            score=$((score - 1))
            issues="${issues} Too short (<50 chars);"
        fi
        
        if ! grep -qi "description" <<< "$content"; then
            score=$((score - 1))
            issues="${issues} No description;"
        fi
        
        if ! grep -qi "usage\|example\|run" <<< "$content"; then
            score=$((score - 1))
            issues="${issues} No usage;"
        fi
        
        if ! grep -qi "install\|setup\|build" <<< "$content"; then
            score=$((score - 1))
            issues="${issues} No installation;"
        fi
        
        if ! grep -qi "fleet\|cocapn\|plato" <<< "$content"; then
            score=$((score - 1))
            issues="${issues} No fleet context;"
        fi
        
        if grep -qi "fleet context" <<< "$content"; then
            score=$((score + 1))
        fi
        
        if grep -qi "deadband" <<< "$content"; then
            score=$((score + 1))
        fi
    fi
    
    echo "$score|$repo_path|$issues"
}

# Run for each repo
for git_dir in $(find /home /mnt /opt /fleet -type d -name ".git" 2>/dev/null); do
    repo_path=$(dirname "$git_dir")
    score_repo "$repo_path"
done | sort -t'|' -k1,1n | head -10
```
