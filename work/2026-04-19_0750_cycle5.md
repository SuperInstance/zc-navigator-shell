### Step 3 — Optimize within channel (P2)  
**Execute scan with scoring:**

```bash
#!/bin/bash
# nav_scan_readme.sh
REPO_LIST=$(find /home -type d -name ".git" 2>/dev/null | grep -v "/.git/" | head -50)
OUTPUT_FILE="/tmp/readme_scores.txt"
echo "Repo,Exists,Length,HasHeaders,HasLinks,TotalScore" > "$OUTPUT_FILE"

for repo in $REPO_LIST; do
    repo_dir=$(dirname "$repo")
    readme=$(find "$repo_dir" -maxdepth 1 -type f -iname "readme*" | head -1)
    exists=0
    length=0
    headers=0
    links=0
    
    if [[ -n "$readme" ]]; then
        exists=1
        # Length score: 1 if > 5 lines
        line_count=$(wc -l < "$readme" 2>/dev/null || echo 0)
        [[ $line_count -gt 5 ]] && length=1
        # Headers: contains at least one '#' line
        grep -q "^#" "$readme" 2>/dev/null && headers=1
        # Links: contains http or markdown link
        grep -q -E "\[.*\]\(.*\)|http" "$readme" 2>/dev/null && links=1
    fi
    
    total=$((exists + length + headers + links))
    echo "\"$repo_dir\",$exists,$length,$headers,$links,$total" >> "$OUTPUT_FILE"
done

# Sort by total score, pick worst 10
echo "=== WORST 10 READMEs ==="
sort -t, -k6,6n "$OUTPUT_FILE" | head -10
```

**Run it:**

```bash
chmod +x nav_scan_readme.sh
./nav_scan_readme.sh
```
