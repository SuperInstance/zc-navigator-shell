```bash
#!/bin/bash
# Score each README
rubric() {
    local repo="$1"
    local score=0
    local readme=""
    
    for f in "README.md" "README.txt" "README"; do
        if [ -f "$repo/$f" ]; then
            readme="$repo/$f"
            break
        fi
    done
    
    if [ -z "$readme" ]; then
        echo "0"  # No README → worst score
        return
    fi
    
    # Heuristic scoring
    score=10  # base
    
    # Length check (too short is bad)
    local lines=$(wc -l < "$readme" 2>/dev/null || echo 0)
    if [ "$lines" -lt 5 ]; then
        score=$((score - 3))
    fi
    
    # Structure check (has headings)
    if ! grep -q "^#" "$readme" 2>/dev/null; then
        score=$((score - 2))
    fi
    
    # Freshness (year in file)
    if ! grep -q "202[4-6]" "$readme" 2>/dev/null; then
        score=$((score - 1))
    fi
    
    # Clarity (has example or code block)
    if ! grep -q -E "```|example|Example" "$readme" 2>/dev/null; then
        score=$((score - 2))
    fi
    
    # Ensure non‑negative
    if [ "$score" -lt 0 ]; then
        score=0
    fi
    
    echo "$score"
}

# Score all repos
output="/tmp/readme_scores.txt"
> "$output"
while IFS= read -r repo; do
    score=$(rubric "$repo")
    echo "$score $repo" >> "$output"
done < /tmp/fleet_repos.txt

# Sort by score (lowest first)
sort -n "$output" | head -20 > /tmp/worst_readmes.txt
```
